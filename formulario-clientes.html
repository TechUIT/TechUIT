Solicitud de Servicio Técnico

Ingresa tus datos para que un experto pueda contactarte.

create table public.clientes ( id bigint generated by default as identity not null, fecha_solicitud timestamp with time zone not null default now(), nombre text null, telefono text null, direccion text null, tipo_servicio text null, problema text null, status text not null default 'Pendiente - Pago Diagnóstico'::text, constraint clientes_pkey primary key (id) ) TABLESPACE pg_default;
